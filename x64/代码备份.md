//递归阶乘
int loop(int x)
{
	for (int a = x; a > 1;)
	{
		return loop(a - 1) * a;
	}
}

int loop_2(int x)
{
	if (x > 1)
		return loop_2(x - 1) * x;
	else
		return 1;
}


//乘法表
void times_sheet()
{
	for (int i = 1; i < 10; i++)
	{
		for (int j = 1; j <= i; j++)
		{
			printf("%dx%d=%d", j, i, j * i);
		}
	}
}


//水仙花
void main()
{
	int hun;
	int ten;
	int alo;
	for (int i = 1; i <= 999;i++)
	{
		hun = i / 100;
		ten = (i / 10) % 10;
		alo = i % 10;
		if (hun*hun*hun + ten*ten*ten + alo*alo*alo == i && i!=1)
		{
			printf("%d = %d^3 + %d^3 + %d^3\n", i, hun, ten, alo);
		}

	}
}
